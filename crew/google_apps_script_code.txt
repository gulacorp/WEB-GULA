function doPost(e) {
  try {
    console.log('doPost called with:', e);
    
    const ss = getSpreadsheet();
    console.log('Spreadsheet accessed:', ss.getId());
    
    let sheet = ss.getSheetByName(SHEET_NAME);
    if (!sheet) {
      console.log('Creating sheet:', SHEET_NAME);
      sheet = ss.insertSheet(SHEET_NAME);
    }

    const body = e.postData && e.postData.contents ? JSON.parse(e.postData.contents) : [];
    console.log('Parsed body:', body);

    // Check if email already exists
    const rows = sheet.getDataRange().getValues();
    console.log('Sheet has', rows.length, 'rows');
    
    const header = rows[0].map(c => c.toString().toLowerCase().trim());
    const emailIdx = header.indexOf('email');
    const email = body[2] || '';
    
    if (email) {
      console.log('Checking for existing email:', email);
      for (let i = 1; i < rows.length; i++) {
        const row = rows[i];
        const rowEmail = (row[emailIdx] || '').toString().trim().toLowerCase();
        console.log('Comparing with row', i, 'email:', rowEmail);
        if (rowEmail === email.toLowerCase()) {
          console.log('Email already exists!');
          // Return existing user data instead of error
          const existingUser = {
            nombre: row[0] || '',
            apellido: row[1] || '',
            email: row[2] || '',
            zip: row[3] || '',
            telefono: row[4] || '',
            puntos: Number(row[5]) || 0,
            instagramFollow: row[6] || '',
            id: row[7] || '',
            fecha: row[8] || ''
          };
          return ContentService
            .createTextOutput(JSON.stringify({ 
              status: 'email_exists', 
              message: 'Email ya registrado',
              existingUser: existingUser
            }))
            .setMimeType(ContentService.MimeType.JSON);
        }
      }
    }

    // Add header row if sheet is empty
    if (sheet.getLastRow() === 0) {
      console.log('Adding header row');
      sheet.appendRow(['Nombre','Apellido','Email','Codigo Postal','Telefono','Puntaje','Instagram Follow','MemberID','Timestamp']);
    }

    // Array order: [NOMBRE, APELLIDO, EMAIL, CODIGO POSTAL, TELEFONO, PUNTAJE, INSTAGRAM, FECHA, USER CODE]
    const row = [
      body[0] || '',  // Nombre
      body[1] || '',  // Apellido
      body[2] || '',  // Email
      body[3] || '',  // Codigo Postal
      body[4] || '',  // Telefono
      body[5] || 0,   // Puntaje
      body[6] || 'NO', // Instagram Follow
      body[8] || '',  // MemberID (USER CODE)
      body[7] || ''   // Timestamp (FECHA)
    ];
    
    console.log('Appending row:', row);
    sheet.appendRow(row);

    console.log('Success! Row added.');
    
    // Send welcome email via Gmail
    const nombre = body[0] || '';
    const memberId = body[8] || '';
    const puntos = Number(body[5]) || 0;
    const fecha = body[7] || new Date().toLocaleDateString('es-ES');
    
    if (email && nombre) {
      try {
        console.log('Attempting to send email to:', email, 'for:', nombre);
        console.log('Email function starting...');
        sendWelcomeEmail(email, nombre, memberId, puntos, fecha);
        console.log('Email function completed successfully');
      } catch (emailErr) {
        console.error('Email sending error (caught):', emailErr.toString());
        console.error('Error message:', emailErr.message);
        console.error('Error stack:', emailErr.stack);
        // Don't fail registration if email fails
      }
    } else {
      console.warn('Skipping email - email or nombre missing. Email:', email, 'Nombre:', nombre);
    }
    
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok' }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    console.error('doPost error:', err);
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// Function to clear all user data from the sheet
function clearAllData() {
  try {
    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName(SHEET_NAME);
    
    if (!sheet) {
      return createJsonOutput({status:'error', message:'Sheet no encontrada'});
    }
    
    const lastRow = sheet.getLastRow();
    console.log('Current rows in sheet:', lastRow);
    
    if (lastRow <= 1) {
      // Only header or empty
      return createJsonOutput({status:'ok', message:'La base de datos ya estaba vacía'});
    }
    
    // Delete all rows except header (rows 2 to lastRow)
    if (lastRow > 1) {
      sheet.deleteRows(2, lastRow - 1);
      console.log('Deleted rows 2 to', lastRow);
    }
    
    console.log('Data cleared successfully');
    return createJsonOutput({
      status:'ok', 
      message:'Base de datos limpiada exitosamente',
      rowsDeleted: lastRow - 1
    });
  } catch (err) {
    console.error('Error clearing data:', err);
    return createJsonOutput({status:'error', message: err.toString()});
  }
}

function getLeaderboard() {
  const ss = getSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    return createJsonOutput({status:'error', message:'Sheet no encontrada'});
  }

  const rows = sheet.getDataRange().getValues();
  if (rows.length <= 1) {
    return createJsonOutput({status:'ok', leaderboard: []});
  }

  const header = rows[0].map(c => c.toString().toLowerCase().trim());
  const nameIdx = header.indexOf('nombre');
  const lastNameIdx = header.indexOf('apellido');
  const emailIdx = header.indexOf('email');
  const pointsIdx = header.indexOf('puntaje');
  const codeIdx = header.indexOf('memberid');

  const users = [];
  for (let i = 1; i < rows.length; i++) {
    const row = rows[i];
    users.push({
      nombre: row[nameIdx] || '',
      apellido: row[lastNameIdx] || '',
      email: row[emailIdx] || '',
      puntos: Number(row[pointsIdx]) || 0,
      id: row[codeIdx] || ''
    });
  }

  // Sort by points descending
  users.sort((a, b) => b.puntos - a.puntos);

  return createJsonOutput({status:'ok', leaderboard: users});
}

  const idIdx = header.indexOf('id');
  const titleIdx = header.indexOf('title');
  const descIdx = header.indexOf('description');
  const pointsIdx = header.indexOf('points');
  const typeIdx = header.indexOf('mission type');
  const activeIdx = header.indexOf('active');

  const missionsByType = {
    'Misiones Semanales': [],
    'Misiones Flash': [],
    'Misiones de Comunidad': []
  };

  for (let i = 1; i < rows.length; i++) {
    const row = rows[i];
    // Mostrar todas las misiones siempre (sin filtrar por Active)
    const missionType = row[typeIdx] || 'Misiones de Comunidad'; // Default to community if not specified

        description: row[descIdx] || '',
        points: Number(row[pointsIdx]) || 0,
        type: missionType

  const dateIdx = header.indexOf('date');
  const locationIdx = header.indexOf('location');
  const pointsIdx = header.indexOf('points');

  return createJsonOutput({status:'ok', events});

function queryMembers(e){
  const ss = getSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    return createJsonOutput({status:'error', message:'Sheet no encontrada'});
  }

  const code = (e.parameter.memberCode || e.parameter.code || '').toString().trim().toUpperCase();
  const email = (e.parameter.email || '').toString().trim().toLowerCase();

  if (!code && !email) {
    return createJsonOutput({status:'ready'});
  }

  const rows = sheet.getDataRange().getValues();
  const header = rows[0].map(c => c.toString().toLowerCase().trim());
  const emailIdx = header.indexOf('email');
  const codeIdx = header.indexOf('memberid');

  const members = [];
  for (let i = 1; i < rows.length; i++) {
    const row = rows[i];
    const rowCode = (row[codeIdx] || '').toString().trim().toUpperCase();
    const rowEmail = (row[emailIdx] || '').toString().trim().toLowerCase();
    if ((code && rowCode === code) || (email && rowEmail === email)) {
      members.push({
        nombre: row[0] || '',
        apellido: row[1] || '',
        email: row[2] || '',
        zip: row[3] || '',
        telefono: row[4] || '',
        puntos: Number(row[5]) || 0,
        instagramFollow: row[6] || '',
        id: row[7] || '',
        fecha: row[8] || ''
      });
    }
  }

  return createJsonOutput({status:'ok', members});
}

}

// Alternative registration via GET parameters (fallback for POST issues)
function registerViaGet(e){
  try {
    // Convert GET parameters to array format expected by doPost
    const data = [
      e.parameter.nombre || '',
      e.parameter.apellido || '',
      e.parameter.email || '',
      e.parameter.zip || '',
      e.parameter.telefono || '',
      parseInt(e.parameter.puntos) || 0,
      e.parameter.instagram || 'NO',
      e.parameter.fecha || '',
      e.parameter.memberId || ''
    ];
    
    const mockEvent = { postData: { contents: JSON.stringify(data) } };
    return doPost(mockEvent);
  } catch (err) {
    return createJsonOutput({status:'error', message: err.toString()});
  }
}
